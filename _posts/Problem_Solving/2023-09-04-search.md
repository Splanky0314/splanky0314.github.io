#include <vector>
using namespace std;

vector<int> v = {1,3,5,7,11,18,21,33,43,45};

int search(int tar, int l, int r) {
    if(l>r) return -1;

    int mid = (l+r)/2;
    if(v[mid]==tar) return mid;
    if(v[mid]<tar) return search(tar, mid+1, r);
    if(v[mid]>tar) return search(tar, l, mid-1);    
}

int main() {
    int target, len=10;
    
    cin >> target;
    cout << search(target, 1, len-1);
}
