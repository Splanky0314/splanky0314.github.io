## "2442번 별 찍기 - 5" 정답 코드
```c
#include <stdio.h>
int main() {
	int n;
	scanf("%d", &n);

	for(int j=0; j<n; j++) {
		for(int i=n-2-j; i>=0; i--) {
			printf(" ");
		}
		for(int i=1; i<=(1+2*(j)); i++) {
			printf("*");
		}
		printf("\n");
	}
}
```
