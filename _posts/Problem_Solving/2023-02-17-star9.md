## "2446번 별 찍기 - 9" 정답 코드
```c
#include <stdio.h>
int main() {
	int n;
	scanf("%d", &n);
	
	for(int i=1; i<=n; i++) {
		for(int j=1; j<=i-1; j++) {
			printf(" ");
		}
		for(int j=1+(n-i)*2; j>0; j--) {
			printf("*");
		}
		printf("\n");
	}
	
	for(int i=n-1; i>=1; i--) {
		for(int j=1; j<=i-1; j++) {
			printf(" ");
		}
		for(int j=1+(n-i)*2; j>0; j--) {
			printf("*");
		}
		printf("\n");
	}
}
```
