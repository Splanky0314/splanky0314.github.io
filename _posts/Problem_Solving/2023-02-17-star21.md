## "10996번 별 찍기 - 21" 정답 코드
```c
#include <stdio.h>
int main() {
	int n;
	scanf("%d", &n);
	
	for(int i=0; i<n*2; i++) {
		if(i%2==0) {
			printf("*");
			for(int j=0; j<(n-1)/2; j++) {
				printf(" *");
			}
		}
		else {
			for(int j=0; j<n/2; j++) {
				printf(" *");
			}
		}
		printf("\n");
	}
}
```
