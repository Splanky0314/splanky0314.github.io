## "2439번 별 찍기 - 2" 정답 코드
```c
#include <stdio.h>
int main() {
	int n;
	scanf("%d", &n);
	for(int i=1; i<=n; i++) {
		for(int j=1; j<=(n-i); j++) {
			printf(" ");
		}
		for(int j=1; j<=i; j++) {
			printf("*");
		}
		printf("\n");
	}                       
}
```
