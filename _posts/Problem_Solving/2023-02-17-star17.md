## "10992번 별 찍기 - 17" 정답 코드
```c
#include <stdio.h>
int main() {
	int n;
	scanf("%d", &n);
	
	for(int i=0; i<n; i++) {
		for(int j=n-i; j>1; j--) {
			printf(" ");
		}
		if(i==(n-1)) {
			for(int j=0; j<1+i*2; j++) printf("*");
			return 0;
		}
		printf("*");
		if(i!=0) {
			for(int j=0; j<1+(i-1)*2; j++) printf(" ");
			printf("*");
		}
		printf("\n");
	}
}
```
