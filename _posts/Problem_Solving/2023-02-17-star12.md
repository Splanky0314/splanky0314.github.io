## "2522번 별 찍기 - 12" 정답 코드
```c
#include <stdio.h>
int main() {
	int n;
	scanf("%d", &n);
	
	for(int i=0; i<n; i++) {
		for(int j=n-i-2; j>=0; j--) printf(" ");
		for(int j=1; j<i+2; j++) printf("*");
		printf("\n");
	}
	for(int i=n-2; i>=0; i--) {
		for(int j=n-i-2; j>=0; j--) printf(" ");
		for(int j=1; j<i+2; j++) printf("*");
		printf("\n");
	}
}
```
