void solve(long long x, long long y, int kk, int idx) {
	int nk = kk+abs(finx-x)+abs(finy-y);
	if(nk <= k) {

		if(ans == -1 || ans > nk)

	nk = kk+2;

		ny = y+v[i].second;

			if(ans == -1 || ans > nk) {
				ans = nk;
			}
			return;
		}
		else {
			// cout << "memo" << nx << " " << ny << " " << nk << endl; //
			//memo[{nx, ny}] = nk;
			solve(nx, ny, nk, i);
		}
	}
}

	}
