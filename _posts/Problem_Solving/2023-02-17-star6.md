## "2443번 별 찍기 - 6" 정답 코드
```c
#include <stdio.h>
int main() {
	int n;
	scanf("%d", &n);

	for(int i=0; i<n; i++) {
		for(int j=0; j<i; j++) {
			printf(" ");
		}
		for(int j=1+((n-i)-1)*2; j>0; j--) {
			printf("*");
		}
		printf("\n");
	} 
}
```
