## "10991번 별 찍기 - 16" 정답 코드
```c
#include <stdio.h>
int main() {
	int n;
	scanf("%d", &n);
	
	for(int i=0; i<n; i++) {
		for(int j=n-i-2; j>=0; j--) {
			printf(" ");
		}
		printf("*");
		for(int j=0; j<i; j++) {
			printf(" *");
		}
		printf("\n");
	}
}
```
