## "10990번 별 찍기 - 15" 정답 코드
```c
#include <stdio.h>
int main() {
	int n;
	scanf("%d", &n);
	
	for(int i=0; i<n; i++) {
		for(int j=n-2-i; j>=0; j--) {
			printf(" ");
		}
		printf("*");
		for(int j=2; j<=i*2; j++) {
			printf(" ");
		}
		if(i) {
			printf("*");
		}
		printf("\n");
	}
}
```
